# 1. Filter for SIP transactions
sip = transactions[transactions["transaction_type"] == "SIP"].copy()

# 2. Convert transaction_date to datetime to enable the .dt accessor
sip["transaction_date"] = pd.to_datetime(sip["transaction_date"])

# 3. Create the aum_crore column from amount_inr (1 crore = 1e7)
sip["aum_crore"] = sip["amount_inr"] / 1e7

# 4. Extract the month period
sip["month"] = (
    sip["transaction_date"]
    .dt.to_period("M")
    .astype(str)
)

# 5. Group and aggregate
monthly_sip = (
    sip.groupby("month")
    ["aum_crore"]
    .sum()
    .reset_index()
)
